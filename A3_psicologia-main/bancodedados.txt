cadastro 
   professores:
	create table professor (id int(3) auto_increment, nome varchar(255), cpf int(11), senha varchar(255), email varchar(255), telefone int (11), nivel varchar(9), primary key(id));

	cadastro para alunos:
	create table aluno(id int(3) auto_increment, nome varchar(255), cpf int(11), ra int(9), senha int(8), email varchar(255), telefone int(11), nivel varchar(9), primary key(id));

    cadastro adm:
	create table adm(id int(3) auto_increment, login varchar(255), senha varchar(255), nivel varchar(9), primary key(id));	


    create table usuarios(id int(3) auto_increment, nome varchar(255), cpf int(11),  ra int(9), senha varchar(255), email varchar(255), telefone int (11), nivel varchar(9), primary key(id));
	

   cadastro de paciente
	create table paciente(id int(3) auto_increment, id_aluno int(3), data_abertura date, nome varchar(255), data_nascimento date, genero varchar(50), endereco text, telefone int(11), email varchar(255), contato_emergencial int(11), escolaridade varchar(255), ocupacao varchar(255), necessidade text, estagiario varchar(255), orientador varchar(255), primary key(id), foreign key(id_aluno) references aluno(id));
